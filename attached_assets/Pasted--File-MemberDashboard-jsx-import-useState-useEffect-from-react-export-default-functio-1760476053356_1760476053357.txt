// File: MemberDashboard.jsx
import { useState, useEffect } from "react";

export default function MemberDashboard() {
  const [user, setUser] = useState(null);
  const [message, setMessage] = useState("");
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Check if user exists in localStorage
    const current = JSON.parse(localStorage.getItem("currentUser") || "null");

    if (current) {
      setUser(current);
      setLoading(false);
    } else {
      // If not found, redirect to join instead of login
      window.location.href = "/join";
    }
  }, []);

  // Still loading or redirecting
  if (loading) {
    return (
      <div className="flex justify-center items-center h-screen">
        <p className="text-gray-600 text-lg">Loading your dashboard...</p>
      </div>
    );
  }

  if (!user) return null;

  const handleDelete = () => {
    if (!window.confirm("Are you sure you want to permanently delete your CoogsNation membership?")) return;

    const users = JSON.parse(localStorage.getItem("coogsnationUsers") || "[]");
    const remaining = users.filter((u) => u.email !== user.email);
    localStorage.setItem("coogsnationUsers", JSON.stringify(remaining));
    localStorage.removeItem("currentUser");

    setMessage("Your membership has been permanently deleted.");
    setTimeout(() => (window.location.href = "/"), 1500);
  };

  return (
    <div className="max-w-4xl mx-auto mt-10 p-6 bg-white shadow rounded-lg">
      <h2 className="text-2xl font-bold text-red-700 mb-2">
        Welcome back, {user.name}!
      </h2>
      <p className="text-gray-700 mb-8">
        You’re logged in as <strong>{user.email}</strong>.
      </p>

      {/* Your dashboard content here */}

      <div className="mt-16 text-center border-t pt-4">
        <button
          onClick={handleDelete}
          className="text-sm text-gray-500 hover:text-red-600"
        >
          ❌ Delete Membership
        </button>
        {message && <p className="mt-2 text-green-600">{message}</p>}
      </div>
    </div>
  );
}
